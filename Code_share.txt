name: Upload Python Files and Run DAG Generator

on:
  push:
    paths:
      - 'config.yaml'

jobs:
  upload-and-run:
    runs-on: ubuntu-latest

    steps:
    - name: Checkout repository
      uses: actions/checkout@v3

    - name: Set up Python
      uses: actions/setup-python@v3
      with:
        python-version: '3.x'

    - name: Install AWS CLI
      run: |
        pip install awscli

    - name: Upload Python files to S3
      env:
        AWS_ACCESS_KEY_ID: ${{ secrets.AWS_ACCESS_KEY_ID }}
        AWS_SECRET_ACCESS_KEY: ${{ secrets.AWS_SECRET_ACCESS_KEY }}
        AWS_REGION: ${{ secrets.AWS_REGION }}
        S3_BUCKET_SOURCE: ${{ secrets.S3_BUCKET_SOURCE }}
      run: |
        aws s3 cp src/ s3://${{ env.S3_BUCKET_SOURCE }}/ --recursive --exclude "*" --include "*.py"

    - name: Run DAG generator script
      run: |
        python path/to/your/dag_generator.py

    - name: Upload DAG file to S3
      env:
        AWS_ACCESS_KEY_ID: ${{ secrets.AWS_ACCESS_KEY_ID }}
        AWS_SECRET_ACCESS_KEY: ${{ secrets.AWS_SECRET_ACCESS_KEY }}
        AWS_REGION: ${{ secrets.AWS_REGION }}
        S3_BUCKET_DAG: ${{ secrets.S3_BUCKET_DAG }}
      run: |
        aws s3 cp path/to/generated_dag_file.py s3://${{ env.S3_BUCKET_DAG }}/
