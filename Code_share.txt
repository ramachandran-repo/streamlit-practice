name: Append Timestamp and Set Env Variable

on: [push]

jobs:
  build:
    runs-on: ubuntu-latest

    steps:
    - name: Checkout code
      uses: actions/checkout@v2

    - name: Create timestamp and append to string
      id: create_timestamp
      run: |
        TIMESTAMP=$(date +"%Y-%m-%d_%H-%M-%S")
        STRING_WITH_TIMESTAMP="YourString_${TIMESTAMP}"
        echo "STRING_WITH_TIMESTAMP=${STRING_WITH_TIMESTAMP}" >> $GITHUB_ENV

    - name: Output the environment variable
      run: echo ${{ env.STRING_WITH_TIMESTAMP }}
