from datetime import datetime

# Get the current date and time
current_datetime = datetime.now()

# Format the date and time as a string suitable for a file name
formatted_datetime = current_datetime.strftime("%Y%m%d_%H%M%S")

# Base file name
base_file_name = "my_file"

# Add the date and time to the file name
file_name_with_datetime = f"{base_file_name}_{formatted_datetime}.txt"

print(file_name_with_datetime)
