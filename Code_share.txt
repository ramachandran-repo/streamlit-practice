def calculate_ecs_resources(data_size_gb):
    # Constants for moderate processing
    CPU_PER_10GB = 2  # 2 vCPUs per 10GB
    MEMORY_BUFFER_PERCENT = 0.25  # 25% buffer

    # Calculate the required vCPUs
    required_vcpus = (data_size_gb / 10) * CPU_PER_10GB

    # Calculate the required memory with buffer
    required_memory_gb = data_size_gb + (data_size_gb * MEMORY_BUFFER_PERCENT)

    # Convert memory to MB (since ECS task definitions use MB)
    required_memory_mb = required_memory_gb * 1024

    return {
        'cpu_units': int(required_vcpus * 1024),  # Convert vCPUs to CPU units
        'memory_mb': int(required_memory_mb)
    }

# Example usage
data_size_gb = 100  # Input data size in GB
resources = calculate_ecs_resources(data_size_gb)

print(f"Required CPU (units): {resources['cpu_units']}")
print(f"Required Memory (MB): {resources['memory_mb']}")
